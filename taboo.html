        <div class="timer" id="timer">90</div>

    </div>


    <script>

        let words = [

            { word: "Flight", forbidden: ["Plane", "Airport", "Travel"] },

            { word: "Computer", forbidden: ["Laptop", "Keyboard", "Screen"] },

            { word: "Dog", forbidden: ["Bark", "Pet", "Animal"] },

            { word: "Football", forbidden: ["Soccer", "Goal", "Kick"] },

            { word: "Phone", forbidden: ["Call", "Number", "Talk"] },

            { word: "Guitar", forbidden: ["Strings", "Music", "Instrument"] },

            { word: "Television", forbidden: ["Screen", "Remote", "Show"] },

            { word: "Car", forbidden: ["Drive", "Wheel", "Engine"] },

            { word: "Book", forbidden: ["Read", "Pages", "Library"] },

            { word: "Table", forbidden: ["Wood", "Furniture", "Legs"] }

        ]; 

        let timer;

        let timeLeft = 90;

        let timerRunning = false;


        function changeWord() {

            let randomIndex = Math.floor(Math.random() * words.length);

            document.querySelector(".word-box").innerText = words[randomIndex].word;

            document.getElementById("forbidden-words").innerHTML = words[randomIndex].forbidden.map(word => `<div>${word}</div>`).join("");

            if (!timerRunning) {

                startTimer();

            }

        }


        function startTimer() {

            timerRunning = true;

            timeLeft = 90;

            document.getElementById("timer").innerText = timeLeft;

            timer = setInterval(function () {

                timeLeft--;

                document.getElementById("timer").innerText = timeLeft;

                if (timeLeft <= 0) {

                    clearInterval(timer);

                    alert("Time's up! Select a new word.");

                    timerRunning = false;

                }

            }, 1000);

        }

    </script>
